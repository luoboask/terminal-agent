### 参数处理增强

```typescript
// 健壮的参数处理
let pointsArray: string[] = [];
if (Array.isArray(points)) {
  pointsArray = points;
} else if (typeof points === 'string') {
  // 按逗号或分号分割
  pointsArray = points.split(/[,,;]/).map(s => s.trim());
} else if (points && typeof points === 'object') {
  // 对象转数组
  pointsArray = Object.values(points).map(String);
}
```
